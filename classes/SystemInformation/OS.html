    def self.os
      return @os if @os
      name = RUBY_PLATFORM
      @os = case name
      when /linux/
        :linux
      when /solaris/
        :solaris
      when /bsd/
        :bsd
      when /darwin/ 
        :darwin
      when /win/
        :windows
      else
        :unknown
      end
    end
